\newcommand{\CharacterExplanation}[2]{

\bigskip

% detokenize:
% https://tex.stackexchange.com/questions/195491/ifthenelse-equal-string-comparison-fails
\begin{tabular}{cc}
\ifthenelse{ \equal{\detokenize{\jtopic}}{\detokenize{katakana}}}{
\raisebox{-.5\height}{\KLETTER{#1}} &
}
\ifthenelse{ \equal{\detokenize{\jtopic}}{\detokenize{hiragana}}}{
\raisebox{-.5\height}{\HLETTER{#1}} &
}
\begin{minipage}{12.5cm}
#2
\end{minipage}\\
\end{tabular}

\bigskip

}
