\newcommand{\CharacterExplanation}[2]{
    \bigskip
    \begin{tabular}{cc}
        \ifthenelse{\equal{Katakana}{\jscript}}{\raisebox{-.5\height}{\KLETTER{#1}}}{}%
        \ifthenelse{\equal{Hiragana}{\jscript}}{\raisebox{-.5\height}{\HLETTER{#1}}}{}%
        & \begin{minipage}{12.5cm}#2\end{minipage} \\
    \end{tabular}
    \bigskip
}
